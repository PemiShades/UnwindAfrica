const VOTE_PRICE = 500.00;
const REST_POINTS_PER_VOTE = 100.00;
let ballot = {};

let voteModalData = { nomineeId: null, nomineeName: null, votePrice: null };

function openVoteModal(nomineeId, nomineeName, votePrice) {
    voteModalData = { nomineeId, nomineeName, votePrice: parseFloat(votePrice) };
    document.getElementById('voteModalTitle').textContent = `Vote for ${nomineeName}`;
    document.getElementById('voteVotesInput').value = '1';
    document.getElementById('votePricePerVote').textContent = `₦${parseFloat(votePrice).toLocaleString()}`;
    document.getElementById('voteTotalPrice').textContent = `₦${parseFloat(votePrice).toLocaleString()}`;
    document.getElementById('voteRestPoints').textContent = `${Math.floor(REST_POINTS_PER_VOTE).toLocaleString()} points`;
    document.getElementById('voteModal').style.display = 'flex';
}

function closeVoteModal() {
    document.getElementById('voteModal').style.display = 'none';
}

function updateVotePrice() {
    const votes = parseInt(document.getElementById('voteVotesInput').value) || 1;
    const pricePerVote = voteModalData.votePrice;
    const totalPrice = votes * pricePerVote;
    const restPoints = votes * REST_POINTS_PER_VOTE;
    
    document.getElementById('voteTotalPrice').textContent = `₦${totalPrice.toLocaleString()}`;
    document.getElementById('voteRestPoints').textContent = `${Math.floor(restPoints).toLocaleString()} points`;
}

function confirmVote() {
    const votes = parseInt(document.getElementById('voteVotesInput').value) || 1;
    if (votes < 1) {
        alert('Please enter a valid number of votes');
        return;
    }
    
    ballot[voteModalData.nomineeId] = {
        name: voteModalData.nomineeName,
        votes: votes
    };
    
    updateBallot();
    closeVoteModal();
    showSuccess(`Added ${votes} vote(s) for ${voteModalData.nomineeName} to your ballot!`);
}

function incrementVote(nomineeId) {
    const input = document.getElementById(`votes-${nomineeId}`);
    input.value = parseInt(input.value) + 1;
    updateBallot();
}

function decrementVote(nomineeId) {
    const input = document.getElementById(`votes-${nomineeId}`);
    if (parseInt(input.value) > 0) {
        input.value = parseInt(input.value) - 1;
        updateBallot();
    }
}

function addVotes(nomineeId, nomineeName) {
    const input = document.getElementById(`votes-${nomineeId}`);
    const votes = parseInt(input.value);
    
    if (votes > 0) {
        ballot[nomineeId] = { name: nomineeName, votes: votes };
        updateBallot();
        input.value = 0;
        
        // Show success notification
        showSuccess(`Added ${votes} vote(s) for ${nomineeName} to your ballot!`);
    } else {
        showWarning('Please select at least 1 vote');
    }
}

function updateBallot() {
    const ballotDiv = document.getElementById('ballotItems');
    const summaryDiv = document.getElementById('ballotSummary');
    
    let totalVotes = 0;
    let totalAmount = 0;
    let html = '';
    
    for (const [nomineeId, data] of Object.entries(ballot)) {
        if (data.votes > 0) {
            totalVotes += data.votes;
            totalAmount += data.votes * VOTE_PRICE;
            html += `
                <div style="display: flex; justify-content: space-between; margin-bottom: 0.5rem; padding: 0.5rem; background: #f8f9fa; border-radius: 4px;">
                    <span style="font-size: 0.9rem;">${data.name}</span>
                    <span style="font-weight: 600;">${data.votes} × ₦${VOTE_PRICE}</span>
                </div>
            `;
        }
    }
    
    if (totalVotes > 0) {
        summaryDiv.style.display = 'block';
        ballotDiv.innerHTML = html;
        document.getElementById('totalVotes').textContent = totalVotes;
        document.getElementById('totalAmount').textContent = `₦${totalAmount.toLocaleString()}`;
        document.getElementById('totalRestPoints').textContent = `${(totalVotes * REST_POINTS_PER_VOTE).toLocaleString()} points`;
    } else {
        summaryDiv.style.display = 'none';
    }
}

function clearBallot() {
    if (Object.keys(ballot).length === 0) {
        showInfo('Your ballot is already empty');
        return;
    }
    
    showConfirm({
        title: 'Clear Ballot?',
        message: 'Are you sure you want to remove all votes from your ballot?',
        confirmText: 'Yes, Clear',
        cancelText: 'Cancel',
        confirmClass: 'danger'
    }).then(confirmed => {
        if (confirmed) {
            ballot = {};
            updateBallot();
            showSuccess('Ballot cleared');
        }
    });
}

    if (totalVotes === 0) {

}

// Countdown Timer

const endDate = new Date('2026-05-24T23:59:59').getTime();

function updateCountdown() {
    const now = new Date().getTime();
    const distance = endDate - now;
    
    if (distance < 0) {
        document.getElementById('countdownTimer').textContent = 'Voting has ended';
        return;
    }
    
    const days = Math.floor(distance / (1000 * 60 * 60 * 24));
    const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
    const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
    const seconds = Math.floor((distance % (1000 * 60)) / 1000);
    
    let countdownText = '';
    if (days > 0) {
        countdownText = `${days} days ${hours} hours`;
    } else if (hours > 0) {
        countdownText = `${hours} hours ${minutes} minutes`;
    } else {
        countdownText = `${minutes} minutes ${seconds} seconds`;
    }
    
    document.getElementById('countdownTimer').textContent = countdownText;
}

// Update countdown immediately and then every second
updateCountdown();
